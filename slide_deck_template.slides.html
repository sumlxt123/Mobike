import numpy as np

plt.rcParams['axes.unicode_minus'] = False

%matplotlib inline


# 
mobike_df = pd.read_csv('mobike_master.csv', index_col=0)

mobike_df['start_time'] = mobike_df['start_time'].apply(lambda x: pd.to_datetime(x))
mobike_df['end_time'] = mobike_df['end_time'].apply(lambda x: pd.to_datetime(x))


# 
order_dict = {'ring_stage': ['inside inner ring', 'inside middle ring', 'inside outer ring', 'outside outer ring'],
              'rate': ['high value user', 'middle value user', 'low value user'],
              'isweekend': ['weekdays', 'weekends'],
              'ispeak': ['peak time', 'non peak time']}

for var in order_dict:
    order = pd.api.types.CategoricalDtype(ordered = True, categories = order_dict[var])
    mobike_df[var] = mobike_df[var].astype(order)

# 
mobike_clean = mobike_df[(((mobike_df.speed>=0) & (mobike_df.speed<=20)) & ((mobike_df.total_time <= 400) | (mobike_df.distance>100)) )]

df_total_time['log_total_time'] = np.log10(df_total_time['total_time'])
df_total_time['log_distance'] = np.log10(df_total_time['distance_km'])
cat_vars = ['isweekend', 'ispeak', 'ring_stage', 'rate']

fig, ax = plt.subplots(ncols=4, nrows=2,figsize=[17,10])
color = sb.color_palette()[0]

for i in range(len(cat_vars)):
    var = cat_vars[i]
    # 
    sb.violinplot(data=df_total_time, x=var, y='log_total_time', ax=ax[0,i], color=color)
    # y
    total_time_ticks = (1, 2, 5, 10, 20, 50, 100, 200, 500)
    ax[0,i].set_yticks(np.log10(total_time_ticks));
    ax[0,i].set_yticklabels(total_time_ticks);
    # x
    if i >= 2:
        xlabels = ax[0,i].get_xticklabels()
        ax[0,i].set_xticklabels(xlabels, rotation=10)

    # 
    sb.violinplot(data=df_total_time, x=var, y='log_distance', ax=ax[1,i], color=color)
